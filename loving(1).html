#定义一个函数用于计算每个选项被选择的次数
my_count = function(choices,labs){
  # 输入的是问卷中的一个变量（即一个问题），以及该问题的选项的标签
  count = rep(0, length(labs))
  for (i in (1:length(labs))) {
    count[i] = length(which(str_detect(choices, labs[i])))
  }
  names(count) = labs
  return(count)
}

#定义性别交叉计数函数
my_count_sex = function(choices0, sex_variable, labs, sex){
  count = rep(0,length(labs))
  choices = choices0[sex_variable == sex]
  for (i in 1:length(count)) {
    count[i] = length(which(str_detect(choices,labs[i])))
  }
  names(count) = labs
  return(count)
}

#定义table的重排列函数
my_table_order = function(labs_order, table){
  for (i in 1:length(labs_order)) {
    iter = which(colnames(table) == labs_order[i])
    if(i==1){table_order = cbind(table[,iter])}else 
    {table_order = cbind(table_order,table[,iter])}
  }
  return(table_order)
}
